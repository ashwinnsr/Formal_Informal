% SLIDE 1: The Problem & Research Question

\begin{frame}{The Problem \& Research Question}

\begin{columns}[T]
\begin{column}{0.65\textwidth}
  \begin{block}{The Context}
    India's Textile \& Apparel sector employs nearly 45 million people, but over \textbf{80\%} work in the informal, unorganized sector. As formal lead firms modernize and increase productivity, what happens to these millions of informal workers?
  \end{block}

  \vspace{10pt}
  \begin{alertblock}{Two Competing Hypotheses}
    \begin{itemize}
      \item \textbf{Displacement:} Formal modernization naturally absorbs informality as the economy develops (Lewis, 1954).
      \item \textbf{Adverse Incorporation:} Formal firms actively exploit informality as a structural subsidy, permanently locking them in (Sanyal, 2007).
    \end{itemize}
  \end{alertblock}
\end{column}
\begin{column}{0.32\textwidth}
  \vspace{5pt}
  \begin{exampleblock}{The Core Question}
    \centering
    Is informality a \textit{dying vestige} of underdevelopment, or a \textit{structural feature} of modern capitalism?
  \end{exampleblock}
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Data \& Methodology}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
  \begin{block}{Integrating Surveys (2010–2024)}
    \begin{itemize}
      \item \textbf{Formal Sector:} Annual Survey of Industries (ASI)
      \item \textbf{Informal Sector:} NSS/ASUSE unorganized enterprises
      \item \textbf{Matched Panel:} $N=152$ state-industry-year observations (28 states, Textiles \& Apparel)
      \item \textbf{Enforcement ($E_s$):} State-level factory inspection ratios
    \end{itemize}
  \end{block}
\end{column}
\begin{column}{0.48\textwidth}
  \begin{exampleblock}{Methodological Approach}
    \begin{itemize}
      \item \textbf{High-Dimensional Fixed-Effects:} State clustered standard errors with Year and NIC controls.
      \item \textbf{Focus on Reduced-Form Tests:} Rather than highly parameterized structural estimation, the analysis relies on robust, falsifiable comparative statics.
      \item \textbf{Rigorous Diagnostics:} Oster bounds for OVB, Breusch-Pagan, Shaprio-Wilk, and 10 robustness checks.
    \end{itemize}
  \end{exampleblock}
\end{column}
\end{columns}

\end{frame}

% SLIDE 4: Finding 1 - Strategic Outsourcing
%==============================================================================
\begin{frame}{Finding 1: Strategic Outsourcing}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
  \begin{block}{Regulatory Arbitrage}
    \textbf{Result:} Formal productivity correlates positively with informal outsourcing intensity ($\beta = 9.07 \times 10^{-4}$, $p=0.043$).

    \vspace{10pt}
    \textbf{The Enforcement Multiplier:}
    This effect is \textbf{4.4$\times$ larger} in high-enforcement states ($p=0.009$) compared to low-enforcement states (not significant).
  \end{block}

  \vspace{10pt}
  \begin{alertblock}{Interpretation}
    Strict labor laws do not protect workers; they incentivize formal lead firms to bypass regulations via subcontracting networks.
  \end{alertblock}
\end{column}

% SLIDE 5: Finding 2 - The Monopsony Null Result
%==============================================================================
\begin{frame}{Finding 2: The Monopsony Null Result}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
  \begin{block}{Zero Wage Pass-Through}
    \textbf{Model:} $\ln w_{\text{inf}} = \beta \ln A_{F} + \delta (E_s \times \ln A_{F})$\\
    \textbf{Result:} $\beta = -0.014$ ($p=0.610$) — a statistical \textbf{zero}.
  \end{block}

  \vspace{10pt}
  \begin{exampleblock}{Why does a non-finding matter?}
    \begin{itemize}
      \item In a competitive labor market, rising productivity *should* bid up wages across the value chain.
      \item The consistent null result across 6 robustness checks proves that formal lead firms possess \textbf{near-perfect monopsony power}.
      \item They capture \textit{all} productivity rents, leaving the informal sector perpetually stagnant.
    \end{itemize}
  \end{exampleblock}
\end{column}

% SLIDE 6: Finding 3 - Structural Persistence

\begin{frame}{Finding 3: Structural Persistence}

\begin{columns}[T]
\begin{column}{0.45\textwidth}
  \begin{block}{The Persistence Test}
    Using an AR(1) dynamic panel design covering 2010 to 2024, informal employment exhibits massive path dependence:
    $$ \beta_1 = 0.848 \quad (p < 0.001) $$

    \vspace{5pt}
    \textit{Note: Nickell (1981) bias deflates this coefficient, meaning true persistence is likely even higher.}
  \end{block}
  \vspace{5pt}
  \begin{alertblock}{Disproving Displacement}
    Informality is not dissolving. It is deeply embedded and structurally locked-in to India's manufacturing architecture.
  \end{alertblock}
\end{column}

\begin{frame}{Conclusion \& Policy Implications}

\begin{columns}[T]
\begin{column}{0.65\textwidth}
  \begin{block}{Reviewing the Evidence}
    The data rejects the Displacement hypothesis. India's textile value chains are characterized by \textbf{Adverse Incorporation}. As formal firms modernise, they deepen—not dissolve—their reliance on precariously paid informal labor.
  \end{block}

  \begin{exampleblock}{Policy Trilemma (from Monte Carlo Simulations)}
    \begin{enumerate}
      \item \textbf{Enforcement Alone:} Merely increasing standard labor law enforcement (\textit{Double Enforcement scenario}) reduces formal jobs but fails to raise informal wages. It just shifts production into the shadows.
      \item \textbf{Direct Wage Floors:} Our simulations show only a direct informal wage floor (+20\%) successfully breaks the monopsony trap without significantly shrinking employment.
      \item \textbf{The Way Forward:} Regulation must shift from policing factory gates to directly empowering the bargaining position of the informal base.
    \end{enumerate}
  \end{exampleblock}
\end{column}
